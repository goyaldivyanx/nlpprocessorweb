Quick Start

python example.py 
import pandas as pd
from nlp_myprocessor import lowercase, remove_stopwords

# Load your messy text data
df = pd.DataFrame({
    "text": [
        "AMAZING Product!!! I LOVE this 😍",
        "Not bad, could be better tho...",
        "Fantastic quality & super fast delivery!"
    ]
})

# One-line preprocessing magic ✨
lowercase(df, "text")
remove_stopwords(df, "text")

print(df["text"].values)
# Output: ['amazing product love', 'bad could better', 
#          'fantastic quality super fast delivery']
